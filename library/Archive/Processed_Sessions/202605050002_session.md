# 🛠 **Obsidian Launch Path Diagnostic (v1.0)**

**Purpose:** Determine the _true_ launch path and runtime backend of Obsidian.

```
obsdiag() {
    local pid main cmd backend
    pid=$(pgrep -n obsidian)

    if [ -z "$pid" ]; then
        echo "❌ Obsidian is not running"
        return
    fi

    cmd=$(ps -p "$pid" -o cmd=)
    backend=$(echo "$cmd" | grep -oE 'ozone-platform=[^ ]+' | cut -d= -f2)

    echo "🧭 Obsidian Diagnostic"
    echo "   PID:      $pid"
    echo "   Command:  $cmd"
    echo "   Backend:  ${backend:-unknown}"

    if echo "$cmd" | grep -q "/home/red/.local/bin/obsidian-x11"; then
        echo "   Launch:   ✔ Wrapper active"
    elif echo "$cmd" | grep -q "/usr/bin/obsidian"; then
        echo "   Launch:   ✖ System binary (override ignored)"
    else
        echo "   Launch:   ⚠ Unknown launcher"
    fi
}
```
